EliminaRepetidos:
	push bp
	mov bp, sp
	
	push efx 
	call nuevaLista;devuelve en efx el puntero al inicio de la nueva lista
	add sp,4; para que era este add?

	push ebx; aux de la lista ppal
	push ecx; sera el auxiliar de la nueva lista
	push edx; aux del primero de la lista ppal
	push eex; aux de nodos

	mov edx, [bp+8]
	mov ebx, [edx+sig]
	

ciclo:
	mov ecx, efx
	cmp ebx, edx;asumiendo a la lista no nula claramente
	JZ fin
	
comparaciones:
	cmp ecx, null
	JZ agrega
	cmp [ecx + valor], [ebx + valor]
	JZ eliminab
	mov ecx, [ecx+sig]
	JMP comparaciones
eliminab:
	mov eex, [ebx + ant]
	mov [eex+sig],[ebx+sig]
	mov eex, [eex+sig]
	mov [eex+ant], [ebx+ant]
	mov ebx, eex
	JMP ciclo
agrega:
	push ecx
	push [ebx+valor]
	call nuevoNodo;Agrega un nodo a ecx con el valor 
	add sp, 8
	JMP ciclo
	
fin:
	push efx
	call cuentaLista;devuelve en eax el numero de elementos de efx (la lista nueva)
	add sp,4

	pop eex
	pop edx
	pop ecx
	pop ebx
	mov bp, sp
	pop bp

	ret
