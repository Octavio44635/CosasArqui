PROTOTIPO
PUSH <base>
PUSH <potencia>
CALL potencia
ADD SP, 8
; devuelve en eax resultado

potencia:     PUSH bp
              MOV bp, sp
              PUSH ebx
              PUSH ecx

              MOV ebx, [bp + 8] ; potencia
              MOV ecx, [bp + 12] ; base

              MOV eax, 1

potencia_loop:CMP ebx, 0         ; si esta elevado a la 0
              JZ potencia_fin

              MUL eax, ecx
              SUB ebx, 1
              JMP potencia_loop

potencia_fin: POP ecx
              POP ebx
              MOV sp, bp
              POP bp

              ret
