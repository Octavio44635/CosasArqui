PROTOTIPO SLEN
PUSH <*char>
CALL slen
ADD sp, 4
; devuelve en ecx la longitud de la cadena de caracteres

terminator equ 0
slen:         PUSH bp
              MOV bp, sp
              PUSH ebx

              CMP [bp + 8], terminator
              JZ slen_cero

              MOV ebx, [bp + 8] ; *char
              ADD ebx, 4 ; apunta al siguiente caracter              

              PUSH ebx
              CALL slen
              ADD sp, 4

              ADD ecx, 1    ; aumento contador
              JMP slen_fin

slen_cero:    MOV ecx, 0

slen_fin:     MOV sp, bp
              POP bp

              RET
