PROTOTIPO
PUSH <base>
PUSH <potencia>
CALL potencia
ADD SP, 8
; devuelve en eax resultado

potencia:     PUSH bp
              MOV bp, sp

              PUSH ebx

              CMP [bp + 8], 0       ; si esta elevado a la 0
              JZ potencia_cero

              MOV ebx, [bp + 8]    ; potencia           
              PUSH [bp + 12]       ; base
              
              SUB ebx, 1   ; resto uno a la potencia
              
              ; recursividad
              PUSH ebx
              CALL potencia
              ADD sp, 8

              MUL eax, [bp + 12]
              JMP potencia_fin

potencia_cero:MOV eax, 1
potencia_fin: POP ebx
              MOV sp, bp
              POP bp

              ret
